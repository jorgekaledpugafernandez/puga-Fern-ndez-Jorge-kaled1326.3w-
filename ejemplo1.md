def max(num1, num2):
    if num1 > num2:
        return num1
    else:
        return num2

# Ejemplo de uso
numero1 = 10
numero2 = 20
resultado = max(numero1, numero2)
print(f"El mayor de {numero1} y {numero2} es {resultado}.")
